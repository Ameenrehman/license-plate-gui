  <ul><li><strong>gui_register.py</strong></li></ul>   
  <ul><li><strong>product2.py</strong></li></ul>
  <ul><li><strong>gui_signup.py</strong></li></ul>
    <p>3. pip install -r requirements.txt </p>
    <p>4. Run main.py</p>
